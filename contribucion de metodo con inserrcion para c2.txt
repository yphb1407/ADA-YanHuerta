// Método para ordenar con Inserción (true = ascendente, false = descendente).


private void ordenarInsercion(double[] arreglo, boolean ascendente) {
    int n = arreglo.length;
    for (int i = 1; i < n; i++) {
        double key = arreglo[i];
        int j = i - 1;

        while (j >= 0 && ((ascendente && arreglo[j] > key) ||
                         (!ascendente && arreglo[j] < key))) {
            arreglo[j + 1] = arreglo[j];
            j--;
        }
        arreglo[j + 1] = key;
    }
}
